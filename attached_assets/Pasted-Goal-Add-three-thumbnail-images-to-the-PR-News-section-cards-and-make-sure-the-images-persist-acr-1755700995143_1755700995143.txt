Component update

Update it to accept imageSrc and imageAlt props and render with next/image using a fixed 3:2 aspect ratio.

Patch example:

// NewsCard.tsx
import Image from "next/image";

type Props = {
  title: string;
  category: string;
  date: string;
  excerpt: string;
  href: string;
  imageSrc: string;   // new
  imageAlt: string;   // new
};

export default function NewsCard({
  title, category, date, excerpt, href, imageSrc, imageAlt
}: Props) {
  return (
    <article className="rounded-2xl shadow-sm bg-white overflow-hidden">
      <a href={href} className="block">
        <div className="relative aspect-[3/2] bg-gray-100">
          <Image
            src={imageSrc}
            alt={imageAlt}
            fill
            sizes="(max-width: 768px) 100vw, 33vw"
            className="object-cover"
            priority={false}
          />
        </div>
        {/* the rest of the card (title, meta, excerpt) stays the same */}
      </a>
    </article>
  );
}
